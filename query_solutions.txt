Selezionare tutti gli eventi, mostrando nome, data di inizio, ora di inizio, ora di fine e durata totale (1040)
    SELECT 
        `name`,
        DATE(`start`) AS `data_inizio`,
        TIME(`start`) AS `ora_inizio`,
        TIME(ADDTIME(`start`, `duration`)) AS `ora_fine`,
        `duration` AS `durata_totale`
    FROM `events`;
